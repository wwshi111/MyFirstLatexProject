\begin{frame}
  \frametitle{图的数学表示}
  
  \begin{columns}
    \column{0.6\textwidth}
    \begin{definition}
      图 $G = (V, E)$ 由顶点集 $V = \{v_1, v_2, \dots, v_n\}$ 和边集 $E \subseteq V \times V$ 组成。
    \end{definition}
    
    \begin{definition}
      邻接矩阵 $\mathbf{A} \in \mathbb{R}^{n \times n}$ 定义为：
      \[
      A_{ij} = 
      \begin{cases} 
      1 & \text{若 } (v_i, v_j) \in E, \\
      0 & \text{否则}.
      \end{cases}
      \]
    \end{definition}

\begin{frame}
  \frametitle{图卷积神经网络基础}
  
  \begin{block}{图卷积的一般形式}
    图卷积层的输出可表示为：
    \[
    \mathbf{H}^{(l+1)} = \sigma\left(\tilde{\mathbf{D}}^{-1/2} \tilde{\mathbf{A}} \tilde{\mathbf{D}}^{-1/2} \mathbf{H}^{(l)} \mathbf{W}^{(l)}\right)
    \]
    其中：
    \begin{itemize}
        \item $\tilde{\mathbf{A}} = \mathbf{A} + \mathbf{I}$（添加自环）
        \item $\tilde{\mathbf{D}}$ 是 $\tilde{\mathbf{A}}$ 的度矩阵
        \item $\mathbf{W}^{(l)}$ 是第 $l$ 层的权重矩阵
        \item $\sigma(\cdot)$ 是非线性激活函数
    \end{itemize}
  \end{block}
  
  \begin{figure}
      \centering
      \begin{tikzpicture}[scale=0.8]
          \node[block] (input) at (0,0) {Input};
          \node[conv] (conv) at (3,0) {GCN Layer};
          \node[block] (output) at (6,0) {Output};
          
          \draw[line] (input) -- node[above] {$\mathbf{H}^{(l)}$} (conv);
          \draw[line] (conv) -- node[above] {$\mathbf{H}^{(l+1)}$} (output);
          \node at (3,-1.5) {$\mathbf{H}^{(l+1)} = \sigma(\tilde{\mathbf{D}}^{-1/2}\tilde{\mathbf{A}}\tilde{\mathbf{D}}^{-1/2}\mathbf{H}^{(l)}\mathbf{W})$};
      \end{tikzpicture}
  \end{figure}
\end{frame}

\begin{frame}
  \frametitle{自注意力机制原理}
  
  \begin{columns}
    \column{0.5\textwidth}
    \begin{block}{标准自注意力计算}
      给定输入序列 $\mathbf{X} = [\mathbf{x}_1, \mathbf{x}_2, \dots, \mathbf{x}_n]$，自注意力计算包含三个步骤：
      
      1. **投影**：计算Query、Key、Value
         \[
         \mathbf{Q} = \mathbf{X}\mathbf{W}_Q, \quad \mathbf{K} = \mathbf{X}\mathbf{W}_K, \quad \mathbf{V} = \mathbf{X}\mathbf{W}_V
         \]
      
      2. **注意力得分计算**：
         \[
         \mathbf{Attention}(Q,K,V) = \text{softmax}\left(\frac{\mathbf{Q}\mathbf{K}^T}{\sqrt{d_k}}\right)\mathbf{V}
         \]
    \end{block}

\begin{frame}
  \frametitle{多头自注意力}
  
  \begin{block}{多头注意力机制}
    为增强模型表达能力，引入多头注意力：
    \[
    \text{MultiHead}(Q,K,V) = \text{Concat}(head_1, head_2, \dots, head_h)\mathbf{W}^O
    \]
    其中每个头的计算为：
    \[
    head_i = \text{Attention}(Q\mathbf{W}_Q^i, K\mathbf{W}_K^i, V\mathbf{W}_V^i)
    \]
  \end{block}

\begin{frame}
  \frametitle{图注意力卷积层设计}
  
  \begin{block}{图注意力卷积公式}
    对于图中的每个节点 $v_i$，其注意力权重计算如下：
    \[
    \alpha_{ij} = \frac{\exp\left(\text{LeakyReLU}\left(\mathbf{a}^T [\mathbf{W}\mathbf{h}_i \| \mathbf{W}\mathbf{h}_j]\right)\right)}
    {\sum_{k \in \mathcal{N}(i)} \exp\left(\text{LeakyReLU}\left(\mathbf{a}^T [\mathbf{W}\mathbf{h}_i \| \mathbf{W}\mathbf{h}_k]\right)\right)}
    \]
    
    其中：
    \begin{itemize}
        \item $\mathbf{a} \in \mathbb{R}^{2d}$ 是注意力权重向量
        \item $\mathbf{W} \in \mathbb{R}^{d \times d}$ 是线性变换矩阵
        \item $\mathcal{N}(i)$ 是节点 $i$ 的邻居集合
        \item $\|\|$ 表示向量拼接操作
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{图结构感知的自注意力机制}
  
  \begin{block}{图感知自注意力计算}
    我们提出一种图结构感知的自注意力机制，其注意力得分计算如下：
    \[
    \mathbf{A}_{att} = \text{softmax}\left(\frac{\mathbf{Q}\mathbf{K}^T}{\sqrt{d_k}} \odot \mathbf{M}\right)
    \]
    
    其中 $\mathbf{M} \in \mathbb{R}^{n \times n}$ 是图结构掩码矩阵，定义为：
    \[
    M_{ij} = 
    \begin{cases} 
    1 & \text{若 } v_j \text{ 在 } v_i \text{ 的 } k\text{-跳邻居内}, \\
    \epsilon & \text{否则（}\epsilon \text{为极小值）}.
    \end{cases}
    \]
  \end{block}
  
  \begin{alertblock}{创新点}
    通过引入图结构掩码，使自注意力机制能够利用图的拓扑信息，限制注意力范围在合理的邻居范围内。
  \end{alertblock}
\end{frame}

\begin{frame}
  \frametitle{特征融合策略}
  
  \begin{block}{双路径特征融合}
    我们采用自适应融合策略整合两条路径的特征：
    \[
    \mathbf{H}_{final} = \lambda \cdot \mathbf{H}_{GAT} + (1-\lambda) \cdot \mathbf{H}_{SA}
    \]
    
    其中自适应权重 $\lambda$ 通过门控机制学习：
    \[
    \lambda = \sigma\left(\mathbf{W}_g [\mathbf{H}_{GAT} \| \mathbf{H}_{SA}] + \mathbf{b}_g\right)
    \]
  \end{block}

        \item 在保证性能的同时保持了较好的效率

% === 9. 讨论 ===
\section{讨论}

\begin{frame}
  \frametitle{方法优势总结}
  
  \begin{block}{核心优势}
    \begin{itemize}
        \item \textbf{表达能力强}: 融合局部和全局信息，提升模型表达能力
        \item \textbf{效率较高}: 通过图结构掩码限制注意力范围，控制复杂度
        \item \textbf{自适应融合}: 门控机制自动学习最优融合策略
        \item \textbf{泛化性好}: 在不同规模和类型的数据集上均表现优异
    \end{itemize}
  \end{block}
  
  \begin{block}{理论贡献}
    \begin{itemize}
        \item 建立了图注意力卷积与自注意力机制的统一框架
        \item 提出了图结构感知的自注意力计算方法
        \item 证明了融合策略的有效性
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{局限性与改进方向}
  
  \begin{block}{当前局限性}
    \begin{itemize}
        \item 计算复杂度仍高于传统GCN
        \item 超参数（如$k$-跳邻居范围）需要手动调整
        \item 在超大规模图上的扩展性仍需提升
    \end{itemize}
  \end{block}
  
  \begin{block}{未来改进方向}
    \begin{itemize}
        \item 探索更高效的注意力计算方法（如稀疏注意力）
        \item 研究自适应邻居范围选择机制
        \item 结合图采样技术提升大规模图处理能力
    \end{itemize}
  \end{block}
\end{frame}
